Задача 1: Программа должна запрашивать возраст пациента до тех пор, пока не будет введен корректный возраст (больше 0 и меньше 120 лет).

python
# Цикл, который будет продолжаться, пока не введен корректный возраст
while True:
    age = int(input("Введите возраст пациента (от 1 до 120 лет): "))
    if 0 < age < 120:
        print(f"Возраст пациента: {age}")

    else:
        print("Некорректный возраст. Попробуйте еще раз.")
