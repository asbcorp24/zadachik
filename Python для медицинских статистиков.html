Задача 1

Создайте переменные для возраста пациента, веса и роста, и вычислите индекс массы тела (ИМТ).

python
# Инициализация переменных
age = 45           # Возраст пациента в годах
weight = 78.5       # Вес пациента в килограммах
height = 1.75       # Рост пациента в метрах

# Расчет ИМТ (Индекс массы тела)
bmi = weight / (height ** 2)  # ИМТ = масса (кг) / (рост (м))^2

# Вывод результата
print(f"Индекс массы тела пациента: {bmi:.2f}")

Комментарии:

  • Определяем переменные для возраста, веса и роста.
  • Вычисляем ИМТ по стандартной формуле.
  • Результат выводится с округлением до двух знаков после запятой.

cholesterol_levels = [5.1, 5.7, 5.3, 6.0, 4.8]  # Уровень холестерина у разных пациентов

def days_until_birthday(birthday_str: str):
    birthday = datetime.datetime.strptime(birthday_str, "%d.%m.%Y")
    today = datetime.datetime.today()
    
    # Проверка, был ли день рождения в этом году
    if today.month > birthday.month or (today.month == birthday.month and today.day > birthday.day):

    else:
        birthday = birthday.replace(year=today.year)

    return (birthday - today).days

python
import torch
import torch.nn as nn
import torch.optim as optim
from torch.utils.data import DataLoader, TensorDataset

# Пример данных: 10 признаков (например, результаты анализов)
X = torch.rand(100, 10)  # 100 пациентов, 10 признаков
y = torch.randint(0, 2, (100,))  # 0 - здоров, 1 - болен

# Датасет и DataLoader
dataset = TensorDataset(X, y)
dataloader = DataLoader(dataset, batch_size=10, shuffle=True)

# Определение модели
class DiseaseModel(nn.Module):
    def __init__(self):
        super(DiseaseModel, self).__init__()
        self.fc1 = nn.Linear(10, 64)
        self.fc2 = nn.Linear(64, 2)  # 2 класса: здоров/болен

    def forward(self, x):
        x = torch.relu(self.fc1(x))
        x = self.fc2(x)
        return x

# Модель и оптимизатор
model = DiseaseModel()
criterion = nn.CrossEntropyLoss()
optimizer = optim.Adam(model.parameters(), lr=0.001)

# Обучение
for epoch in range(10):
    for inputs, labels in dataloader:
        optimizer.zero_grad()
        outputs = model(inputs)
        loss = criterion(outputs, labels)
        loss.backward()
        optimizer.step()
    print(f"Epoch {epoch+1}, Loss: {loss.item()}")

]

def update_patient(name: str, patient: Patient):
    for p in patients:

            return {"message": "Диагноз обновлен"}
    return {"message": "Пациент не найден"}
